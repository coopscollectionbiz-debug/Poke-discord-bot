import { SlashCommandBuilder, EmbedBuilder } from 'discord.js';
import { grantTP } from '../bot_final_integrated.js';
import pokemonData from '../data/pokemonData.json' assert { type: 'json' };
import trainerSprites from '../data/trainerSprites.json' assert { type: 'json' };

export default {
  data: new SlashCommandBuilder()
    .setName('recruit')
    .setDescription('Recruit a random Pok√©mon or Trainer using Coop Coins.')
    .addStringOption(opt =>
      opt.setName('type')
        .setDescription('Recruit a Pok√©mon or Trainer')
        .setRequired(true)
        .addChoices(
          { name: 'Pok√©mon (100 CC)', value: 'pokemon' },
          { name: 'Trainer (200 CC)', value: 'trainer' }
        )
    ),
  async execute(interaction, trainerData) {
    const userId = interaction.user.id;
    const choice = interaction.options.getString('type');

    if (!trainerData[userId]) trainerData[userId] = { tp: 0, cc: 0, pokemon: {}, trainers: {} };
    const user = trainerData[userId];

    const cost = choice === 'pokemon' ? 100 : 200;
    if (user.cc < cost)
      return interaction.reply({ content: `‚ùå You need ${cost} CC to recruit ${choice}.`, ephemeral: true });

    user.cc -= cost;

    // Pok√©mon recruit
    if (choice === 'pokemon') {
      const pool = Object.entries(pokemonData);
      const [id, data] = pool[Math.floor(Math.random() * pool.length)];
      user.pokemon[id] = (user.pokemon[id] || 0) + 1;

      await grantTP(interaction, userId, 50);

      const embed = new EmbedBuilder()
        .setColor(0x00AE86)
        .setTitle(`üéâ You recruited ${data.name}!`)
        .setDescription(`Rarity: ${data.rarity}\nType: ${data.type.join(' / ')}`)
        .setImage(`${data.sprite}`)
        .setFooter({ text: `${interaction.user.username} ‚Ä¢ +50 TP ‚Ä¢ -${cost} CC` });

      return interaction.reply({ embeds: [embed], ephemeral: true });
    }

    // Trainer recruit
    const pool = Object.entries(trainerSprites);
    const [id, data] = pool[Math.floor(Math.random() * pool.length)];
    user.trainers[id] = (user.trainers[id] || 0) + 1;

    await grantTP(interaction, userId, 100);

    const embed = new EmbedBuilder()
      .setColor(0x00AE86)
      .setTitle(`üßë‚Äçüè´ You recruited ${data.name}!`)
      .setImage(`${data.sprite}`)
      .setFooter({ text: `${interaction.user.username} ‚Ä¢ +100 TP ‚Ä¢ -${cost} CC` });

    return interaction.reply({ embeds: [embed], ephemeral: true });
  }
};
